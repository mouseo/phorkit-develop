import { Playground } from 'docz';
import { renderFromProp, renderFromPropWithFallback } from '../index';

## renderFromProp

### As a function

<Playground>{() => renderFromProp(() => <div>Hello world</div>)}</Playground>

### As a function with props

<Playground>{() => renderFromProp(({ recipient }) => <div>Hello {recipient}</div>, { recipient: 'world' })}</Playground>

### As a function with children

<Playground>
  {() => renderFromProp(({ children }) => <div>{children}</div>, {}, { children: 'Hello world' })}
</Playground>

### As a fragment with a function with children

<Playground>
  {() =>
    renderFromProp(
      <React.Fragment>{({ children }) => <div>{children}</div>}</React.Fragment>,
      {},
      { children: 'Hello world' },
    )
  }
</Playground>
